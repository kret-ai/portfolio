# Implementacja sortowania bąbelkowego w Pythonie
# Złożoność czasowa: O(n^2)

def bubble_sort(arr):
    n = len(arr)
    
    # Przejdź przez wszystkie elementy tablicy
    for i in range(n):
        
        # Ostatnie i elementów jest już na swoim miejscu
        for j in range(0, n-i-1):
            
            # Zamień, jeśli element jest większy od następnego
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
                
    return arr

# Przykładowe użycie
data = [64, 34, 25, 12, 22, 11, 90]
sorted_data = bubble_sort(data)
print("Posortowana tablica:", sorted_data)
